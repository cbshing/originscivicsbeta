civic_cbshing_lost_colony = {
	modification = no
	potential = { ethics = { NOT = { value = ethic_gestalt_consciousness } } }
	possible = {
		civics = {
			NOR = {
				value = civic_oc_exiled_colony # base OC
				
				# value = civic_oc_sublight_empire # OC Unification War Addon

				value = civic_oc_fallen_colony  # OC Fallen Dreams Addon
				# value = civic_oc_fallen_enclave # OC Fallen Dreams Addon
				
				# value = civic_oc_post_war_of_worlds # OC Unification War Addon, also incompatible with Colonies civics
				# value = civic_oc_sublight_conquerer # OC Unification War Addon, also incompatible with Colonies civics
				# value = civic_oc_refugee_saviour # OC Unification War Addon, also incompatible with Colonies civics
			}
		}
	}
	random_weight = {
		base = 0
		modifier = {
			# civic has no effect after galaxy generation, disable for random generation
			factor = 0
			has_global_flag = game_started
		}
	}
	modifier = { 
		planet_clear_blocker_cost_mult = -0.10 
		planet_colony_development_speed_mult = 0.10
	}

	#description = "civic_tooltip_oc_lost_colony_effects"
	#used when starting a new game to initialize home planet
	#see 01_origins_civics_empire_setup.5
}

civic_oc_exiled_colony = {
	modification = no
	potential = { ethics = { NOT = { value = ethic_gestalt_consciousness } } }
	possible = {
		civics = {
			NOR = {
				value = civic_cbshing_lost_colony # base OC
				
				# value = civic_oc_sublight_empire # OC Unification War Addon

				value = civic_oc_fallen_colony  # OC Fallen Dreams Addon
				# value = civic_oc_fallen_enclave # OC Fallen Dreams Addon
				
				# value = civic_oc_post_war_of_worlds # OC Unification War Addon, also incompatible with Colonies civics
				# value = civic_oc_sublight_conquerer # OC Unification War Addon, also incompatible with Colonies civics
				# value = civic_oc_refugee_saviour # OC Unification War Addon, also incompatible with Colonies civics
			}
		}
	}
	random_weight = {
		base = 0
		modifier = {
			# civic has no effect after galaxy generation, disable for random generation#
			# can only be generated by event after galaxy generation
			factor = 0
			has_global_flag = game_started
		}
	}
	modifier = { 
		pop_government_ethic_attraction = 0.10 
		faction_happiness = 0.05
		planet_colony_development_speed_mult = 0.10
	}

	#description = "civic_tooltip_oc_exiled_colony_effects"
	#used when starting a new game to initialize home planet
	#see 01_origins_civics_empire_setup.5
}

civic_oc_fallen_colony = {
	modification = no
	playable = { host_has_dlc = "OC_INVALID" }
	potential = { ethics = { NOT = { value = ethic_gestalt_consciousness } } }
	possible = {
		civics = {
			NOR = {
				value = civic_cbshing_lost_colony # base OC
				value = civic_oc_exiled_colony # base OC
				
				# value = civic_oc_sublight_empire # OC Unification War Addon

				# value = civic_oc_fallen_enclave # OC Fallen Dreams Addon
				
				# value = civic_oc_post_war_of_worlds # OC Unification War Addon, also incompatible with Colonies civics
				# value = civic_oc_sublight_conquerer # OC Unification War Addon, also incompatible with Colonies civics
				# value = civic_oc_refugee_saviour # OC Unification War Addon, also incompatible with Colonies civics
			}
		}
	}
	random_weight = {
		# will be generated by event only (unless player picks)
		base = 0
		modifier = {
			# civic has no effect after galaxy generation, disable for random generation
			factor = 0
			has_global_flag = game_started
		}
	}
	modifier = { 
		all_technology_research_speed = 0.05
		pop_consumer_goods_add = 0.15
	}

	#description = "civic_tooltip_oc_fallen_colony_effects"
	#used when starting a new game to initialize home planet
	#see 01_origins_civics_empire_setup.5
}
