	random_weight = {
		factor = 0
		modifier = {
			add = 2
			has_government = "gov_fallen_colonial_government"
			is_pirate = no
			is_primitive = no
			NOT = { is_country_type = fallen_empire }
			NOT = { is_country_type = awakened_fallen_empire }
		}

	prefix_format = "[This.Capital.System.GetName] <oc_fallen_colonial_gen>"
