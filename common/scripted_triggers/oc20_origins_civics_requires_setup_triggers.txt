# Origins Civics Requires Country Setup Triggers
# by cbshing
#
# These triggers are referenced in the 01_origins_civics_empire_setup.txt
# These triggers are used to setup the galaxy.
#
#	Example:
# 
#	example_trigger = {
#		is_country_type = default
#		free_leader_slots > 0
#	}
#
#
#	In a script file:
#
#	trigger = {
#		example_trigger = yes
#	}
#
